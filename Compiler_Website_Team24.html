CFG Context Free Grammar

Defines the formal grammar for the Do-While statement. Non-terminals (green) represent structural components; terminals (yellow) are actual tokens.

DoWhileStmt → do { StatementList } while ( Condition ) ;
StatementList → Statement StatementList | ε
Statement → PrintfStmt | Assignment
Assignment → IDENTIFIER = Expression ; | IDENTIFIER ++ ; | IDENTIFIER -- ;
PrintfStmt → printf ( " STRING " , IDENTIFIER ) ;
Condition → Expression RelOp Expression
RelOp → < | > | == | != | <= | >=
Expression → Term | Expression + Term | Expression - Term
Term → Factor | Term * Factor | Term / Factor
Factor → IDENTIFIER | NUMBER

04 Assembly Code Generator (x86 NASM)

Dynamically reads variable names, initial values, and the comparison operator from the lexeme table — not hardcoded. Works for any valid do-while input.

assembly.txt 
; ============================================
; Assembly — Do-While Compiler Project
; ============================================

section .data
    fmt db "%d", 10, 0

section .bss
    i resd 1          ; int i

section .text
    global main
    extern printf

main:
    ; i = 1
    mov eax, 1
    mov [i], eax

do_loop:
    ; printf("%d\n", i)
    mov eax, [i]
    push eax
    push fmt
    call printf
    add esp, 8

    ; i++
    mov eax, [i]
    add eax, 1
    mov [i], eax

    ; while (i < 5)
    mov eax, [i]
    cmp eax, 5
    jl  do_loop

    ; exit
    mov eax, 0
    ret
